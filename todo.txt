function point2url(point, panoramaProvider = 0) {
	let panoId
	if (point.panoId !=null) {
		if (regexp_ascii.test(point.panoId)) {
			panoId = hex2ascii(point.panoId)
		} else {
			panoId = point.panoId
		}
	}
	let fov = 90
	if (point.zoom !=null) {
		fov = 180 / Math.pow(2, Math.min(Math.max(point.zoom, 0), 4));
	}
	if (point.panoId != null && panoramaProvider == 1) {
		return "https://www.mapillary.com/app/?lat=" + point.lat + "&lng=" + point.lng + "&z=20&pKey=" + panoId + "&focus=photo"
	} else if (point.panoId != null && panoramaProvider == 0) {
		return "https://www.google.com/maps/@?api=1&map_action=pano&pano=" + panoId + "&viewpoint=" + point.lat + "," + point.lng + "&heading=" + point.heading + "&pitch=" + point.pitch + "&fov=" + fov
	} else {
		return "https://www.google.com/maps/@?api=1&map_action=pano&viewpoint=" + point.lat + "," + point.lng + "&heading=" + point.heading + "&pitch=" + point.pitch + "&fov=" + fov
	}
}
